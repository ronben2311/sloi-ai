Non-negotiables
⚡ Boss approval on every LOI
No LOI is ever generated without explicit Boss action. The agent stops at AWAIT_HUMAN and waits. This is hardcoded — cannot be bypassed by any role or agent.
CRITICAL
🔒 Supplier anonymization via RLS
Factory identity is never exposed to buyers. Enforced at DB level (Supabase RLS). Buyers see REF-MET-001 etc. only. Never expose real supplier names in any API response to buyers.
CRITICAL
🛡️ Compliance screening on every deal
Every buyer, supplier, and entity is screened before negotiation starts. Blocked deals (OFAC etc.) cannot proceed. Not optional — runs automatically on deal creation.
CRITICAL
💳 Credits checked before every agent action
Deduct credits atomically before running agent. Use Supabase RPC decrement_credits — never plain UPDATE. Fail the action if balance insufficient.
HIGH
Deploy /.well-known/agent.json before launch
A2A-compatible buyer agents discover SLOI AI via this file. Deploy to Vercel as static file in .well-known/ folder. See agent.json + a2a.html.

1.1 — Database Schema (Supabase)
Create all tables in order. Foreign keys require users → organizations → then all others. Enable RLS on every table.
SQL — run in Supabase SQL editor
-- 1. Organizations
CREATE TABLE organizations (
  id          UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name        TEXT NOT NULL,
  type        TEXT CHECK(type IN ('buyer','broker','supplier','boss')),
  country     TEXT,
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- 2. Users (extends Supabase auth.users)
CREATE TABLE users (
  id          UUID REFERENCES auth.users(id) PRIMARY KEY,
  org_id      UUID REFERENCES organizations(id),
  role        TEXT CHECK(role IN ('buyer','broker','supplier','boss')),
  name        TEXT,
  email       TEXT,
  phone       TEXT,
  wallet_address TEXT,  -- Base wallet for USDC
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- 3. Credits
CREATE TABLE credits (
  user_id         UUID PRIMARY KEY REFERENCES users(id),
  balance         INT NOT NULL DEFAULT 0,
  total_purchased INT DEFAULT 0,
  total_spent     INT DEFAULT 0,
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE credit_transactions (
  id                UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id           UUID REFERENCES users(id),
  credits           INT NOT NULL,  -- positive=purchase, negative=usage
  source            TEXT,            -- stripe|usdc|usage
  action            TEXT,            -- deal_hunter|loi|autopilot|watcher
  amount_usd        NUMERIC(10,2),
  stripe_session_id TEXT,
  tx_hash           TEXT,
  wallet_address    TEXT,
  created_at        TIMESTAMPTZ DEFAULT NOW()
);

-- Atomic credit operations
CREATE OR REPLACE FUNCTION increment_credits(p_user_id UUID, p_amount INT)
RETURNS VOID AS $$
  INSERT INTO credits(user_id, balance, total_purchased)
  VALUES(p_user_id, p_amount, p_amount)
  ON CONFLICT(user_id) DO UPDATE SET
    balance=credits.balance+p_amount, total_purchased=credits.total_purchased+p_amount, updated_at=NOW();
$$ LANGUAGE sql;

CREATE OR REPLACE FUNCTION decrement_credits(p_user_id UUID, p_amount INT)
RETURNS VOID AS $$
  UPDATE credits SET balance=balance-p_amount, total_spent=total_spent+p_amount, updated_at=NOW()
  WHERE user_id=p_user_id AND balance>=p_amount;
  IF NOT FOUND THEN RAISE EXCEPTION 'insufficient_credits'; END IF;
$$ LANGUAGE plpgsql;

-- 4. Negotiations
CREATE TABLE negotiations (
  id              TEXT PRIMARY KEY,        -- NEG-001
  product         TEXT NOT NULL,
  qty             NUMERIC(12,2) NOT NULL,
  unit            TEXT DEFAULT 'MT',
  buyer_id        UUID REFERENCES users(id),
  buyer_name      TEXT,
  broker_id       UUID REFERENCES users(id),
  supplier_ref    TEXT,              -- REF-MET-001 (never real name)
  target_price    NUMERIC(12,2),
  max_price       NUMERIC(12,2),
  strategy        TEXT DEFAULT 'standard',
  status          TEXT DEFAULT 'running',
  current_round   INT DEFAULT 0,
  max_rounds      INT DEFAULT 5,
  agent_price     NUMERIC(12,2),
  broker_price    NUMERIC(12,2),
  deal_price      NUMERIC(12,2),
  boss_decision   TEXT,
  override_price  NUMERIC(12,2),
  loi_ref         TEXT,
  credits_used    INT DEFAULT 0,
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);

-- 5. LOIs
CREATE TABLE lois (
  ref             TEXT PRIMARY KEY,    -- SL-LOI-20260501-001
  neg_id          TEXT REFERENCES negotiations(id),
  product         TEXT,
  qty             NUMERIC(12,2),
  unit            TEXT,
  deal_price      NUMERIC(12,2),
  total_value     NUMERIC(14,2),
  buyer_name      TEXT,
  supplier_ref    TEXT,
  loi_fee_usd     NUMERIC(10,2) DEFAULT 500,
  loi_fee_paid    BOOLEAN DEFAULT false,
  pdf_url         TEXT,
  signed_at       TIMESTAMPTZ DEFAULT NOW()
);

-- 6. Products (supplier catalog, anonymized)
CREATE TABLE products (
  ref             TEXT PRIMARY KEY,    -- REF-MET-001
  supplier_id     UUID REFERENCES users(id),
  sector          TEXT,
  name            TEXT,               -- visible to buyers
  sku             TEXT,
  unit            TEXT,
  current_price   NUMERIC(12,2),
  moq             INT,
  lead_time_days  INT,
  certs           TEXT[],
  active          BOOLEAN DEFAULT true,
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);

-- 7. API keys (for Open Network agents)
CREATE TABLE api_keys (
  id              UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id         UUID REFERENCES users(id),
  key_hash        TEXT NOT NULL UNIQUE,  -- bcrypt hash of sk-sloi-...
  name            TEXT,
  tier            TEXT DEFAULT 'read',     -- read|transact
  wallet_address  TEXT,
  last_used_at    TIMESTAMPTZ,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- 8. Price watches
CREATE TABLE price_watches (
  id              UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id         UUID REFERENCES users(id),
  product_ref     TEXT REFERENCES products(ref),
  target_price    NUMERIC(12,2),
  direction       TEXT DEFAULT 'below',
  triggered       BOOLEAN DEFAULT false,
  active          BOOLEAN DEFAULT true,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- RLS policies (run after table creation)
ALTER TABLE negotiations ENABLE ROW LEVEL SECURITY;
CREATE POLICY neg_buyer   ON negotiations FOR SELECT USING(buyer_id=auth.uid());
CREATE POLICY neg_broker  ON negotiations FOR SELECT USING(broker_id=auth.uid());
CREATE POLICY neg_boss    ON negotiations FOR ALL USING((SELECT role FROM users WHERE id=auth.uid())='boss');
ALTER TABLE products ENABLE ROW LEVEL SECURITY;
CREATE POLICY prod_public ON products FOR SELECT USING(active=true); -- ref only, no supplier_id
CREATE POLICY prod_supplier ON products FOR ALL USING(supplier_id=auth.uid());
