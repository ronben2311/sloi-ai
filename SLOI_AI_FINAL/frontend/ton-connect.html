// ── TON CONNECT ──────────────────────────────────────────────────────────────
class SloiTON {
  constructor() {
    this.connector = new TONConnectUI.TonConnectUI({
      manifestUrl: 'https://sloiai.com/tonconnect-manifest.json',
      buttonRootId: 'ton-connect-button',  // optional: auto-render button
    });

    this.connector.onStatusChange(wallet => {
      if (wallet) {
        this.wallet = wallet;
        this.onConnect(wallet);
      } else {
        this.wallet = null;
        this.onDisconnect();
      }
    });
  }

  async connect() {
    if (!this.connector.connected) {
      await this.connector.openModal();
    }
    return this.wallet;
  }

  onConnect(wallet) {
    const addr = wallet.account.address;
    const short = addr.slice(0, 6) + '...' + addr.slice(-4);
    showToast('💎 Wallet connected: ' + short, 'teal');
    // Save to SLOI AI account
    fetch('/v1/users/wallet', {
      method: 'POST',
      headers: { 'x-api-key': SloiAPI.apiKey },
      body: JSON.stringify({ wallet_address: addr, chain: 'ton' }),
    });
    // Update UI
    document.querySelectorAll('[data-ton-address]').forEach(el => el.textContent = short);
    document.querySelectorAll('[data-ton-connected]').forEach(el => el.style.display = 'flex');
  }

  onDisconnect() {
    document.querySelectorAll('[data-ton-connected]').forEach(el => el.style.display = 'none');
  }

  // Send TON payment for credits
  async buyCredits(pack) {
    const packs = {
      99:  { ton: '9900000000',  credits: 100  },  // nanoTON
      299: { ton: '29900000000', credits: 350  },
      599: { ton: '59900000000', credits: 800  },
      999: { ton: '99900000000', credits: 1500 },
    };
    const p = packs[pack];
    if (!p) return;

    if (!this.connector.connected) await this.connect();

    const tx = {
      validUntil: Math.floor(Date.now() / 1000) + 600,  // 10 min
      messages: [{
        address: 'EQ...',  // SLOI_TON_WALLET (from ENV)
        amount: p.ton,
        payload: buildPayload({ credits: p.credits, user_id: this._getUserId() }),
      }]
    };

    try {
      await this.connector.sendTransaction(tx);
      showToast('💎 Payment sent · Credits appear in ~2 seconds', 'green');
    } catch(e) {
      if (e.message === 'User rejected the transaction') {
        showToast('Payment cancelled', 'red');
      }
    }
  }

  buildPayload({ credits, user_id }) {
    // Encode user_id + pack in transaction comment
    // Backend reads this from webhook to match payment to user
    const comment = `SLOI:${user_id}:${credits}`;
    const bytes = new TextEncoder().encode(comment);
    return btoa(String.fromCharCode(...bytes));
  }

  _getUserId() {
    const user = JSON.parse(localStorage.getItem('sloi_user') || '{}');
    return user.id || 'unknown';
  }

  async disconnect() {
    await this.connector.disconnect();
  }
}

window.SloiTON = SloiTON;

// POST /webhooks/ton — from GetBlock or Tonhub monitor
app.post('/webhooks/ton', async (req, res) => {
  const { from_address, to_address, amount, comment, tx_hash } = req.body;

  // Verify payment is to our wallet
  if (to_address !== process.env.SLOI_TON_WALLET) return res.json({ ok: true });

  // Parse comment: "SLOI:{user_id}:{credits}"
  let userId, credits;
  try {
    const decoded = Buffer.from(comment, 'base64').toString();
    if (decoded.startsWith('SLOI:')) {
      [, userId, credits] = decoded.split(':');
      credits = parseInt(credits);
    }
  } catch {}

  // Fallback: match by wallet address registered to user
  if (!userId) {
    const user = await supabase
      .from('users').select('id')
      .eq('wallet_address', from_address)
      .single();
    userId = user?.data?.id;
  }

  // Determine pack from amount (nanoTON)
  if (!credits) {
    const tonAmount = amount / 1e9;
    const packMap = { 9.9: 100, 29.9: 350, 59.9: 800, 99.9: 1500 };
    credits = packMap[parseFloat(tonAmount.toFixed(1))];
  }

  if (!userId || !credits) return res.json({ ok: true });

  // Add credits atomically
  await supabase.rpc('increment_credits', {
    p_user_id: userId,
    p_amount: credits,
  });

  // Log transaction
  await supabase.from('credit_transactions').insert({
    user_id: userId,
    credits,
    source: 'ton',
    tx_hash,
    wallet_address: from_address,
    amount_usd: credits,  // approximate
  });

  // Notify user via AG-UI stream
  eventBus.publish({
    type: 'STATE_PATCH',
    user_id: userId,
    data: { ops: [{ op: 'replace', path: '/credits', value: 'balance_refreshed' }] }
  });

  res.json({ ok: true });
});
