// Automatic Alerts — Boss receives these
These events are pushed automatically from the AG-UI stream → OpenClaw → Telegram. No command needed.
⚡ AWAIT_HUMAN — deal needs approval
Fires when agent + broker reach agreement. Boss must reply "approve" or "reject".
✅ LOI_GENERATED
Fires after Boss approves. Includes LOI ref, deal value, parties.
🔥 LEAD_FOUND (high/very-high only)
LeadFinder found a relevant opportunity. Boss replies: pipeline / outreach / skip.
📡 PRICE_ALERT
A watched price hit its target. Fires when PriceScout detects threshold crossed.
🛡️ COMPLIANCE_BLOCKED
A deal was blocked by sanctions screening. Deal stopped automatically — no action needed.

// Webhook — SLOI AI → OpenClaw
NODE.JS — SLOI AI backend (routes/stream.js)
const OPENCLAW_WEBHOOK = process.env.OPENCLAW_WEBHOOK_URL;
const ALERT_EVENTS = [
  'AWAIT_HUMAN', 'LOI_GENERATED', 'NEGOTIATION_FAILED',
  'COMPLIANCE_BLOCKED', 'PRICE_ALERT'
];

async function notifyOpenClaw(event) {
  if (!ALERT_EVENTS.includes(event.type)) return;
  if (!OPENCLAW_WEBHOOK) return;

  await fetch(OPENCLAW_WEBHOOK, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      type:            event.type,
      session_id:      event.session_id,
      message:         formatAlertMessage(event),
      requires_action: event.type === 'AWAIT_HUMAN',
      data:            event.data,
    })
  });
}

// Call in eventBus.publish() after routing to portal subscribers
eventBus.publish = function(event, targetRoles) {
  this.subscribers.forEach(({ role, cb }, userId) => {
    if (role === 'boss' || targetRoles.includes(role) || event.user_id === userId) {
      cb(event);
    }
  });
  notifyOpenClaw(event); // ← add this line
};
// Orchestration Rules (hardcoded)
1
Boss approval gate cannot be bypassed
OpenClaw can start, stop, and configure all agents. But it cannot approve LOIs on its own. Every "approve" must come from a Boss message in the current conversation.
2
Compliance blocks are final
OpenClaw cannot override BLOCKED verdicts (OFAC/EU/UN). Only REVIEW verdicts can be manually approved by Boss. This is enforced at API level — 451 error cannot be bypassed.
3
Credit check before every agent start
OpenClaw checks balance before triggering any paid agent action. Alerts Boss if balance < 20 credits before starting negotiation.
4
AWAIT_HUMAN queue — one at a time
If multiple deals need approval simultaneously, OpenClaw queues them and presents one at a time. Boss cannot accidentally approve the wrong deal.
5
Language is routing, not privilege
Hebrew commands route to the same API calls as English. Natural language is interpreted by Claude inside OpenClaw — it never grants extra permissions.
