// Bot Code — routes/telegram.js

const { Telegraf, Markup } = require('telegraf');
const bot = new Telegraf(process.env.TELEGRAM_BOT_TOKEN);
const BOSS_ID = parseInt(process.env.TELEGRAM_BOSS_CHAT_ID);
const CHANNEL = process.env.TELEGRAM_CHANNEL_ID; // @SloiAIExchange
const MINI_APP_URL = 'https://sloiai.com/telegram-mini-app.html';

// ── BOSS AUTH MIDDLEWARE ─────────────────────────────────────────────
const bossOnly = async (ctx, next) => {
  if (ctx.from?.id !== BOSS_ID) {
    return ctx.reply('⛔ Boss only command.');
  }
  return next();
};

// ── COMMANDS ─────────────────────────────────────────────────────────
bot.command('start', (ctx) => ctx.reply(
  '🤖 SLOI AI\nSmart negotiation · LOI generation\n\nCommands: /help\nApp: /app',
  Markup.inlineKeyboard([
    [Markup.button.webApp('📱 Open SLOI AI', MINI_APP_URL)],
    [Markup.button.callback('📊 Prices', 'prices'), Markup.button.callback('📄 My LOIs', 'lois')],
  ])
));

bot.command('app', (ctx) => ctx.reply(
  'Open SLOI AI:',
  Markup.inlineKeyboard([[Markup.button.webApp('📱 SLOI AI Exchange', MINI_APP_URL)]])

bot.command('approve', bossOnly, async (ctx) => {
  const negId = ctx.message.text.split(' ')[1] || await getPendingNegId();
  if (!negId) return ctx.reply('No pending deals found.');
  await approveDeal(negId);
  ctx.reply(`✅ Deal ${negId} approved. LOI generating...`);
});

bot.command('reject', bossOnly, async (ctx) => {
  const negId = ctx.message.text.split(' ')[1] || await getPendingNegId();
  await rejectDeal(negId);
  ctx.reply(`✗ Deal ${negId} rejected.`);
});

bot.command('status', bossOnly, async (ctx) => {
  const s = await getPlatformStatus();
  ctx.reply(

bot.command('prices', async (ctx) => {

  const msg = prices.map(p => {
    const arrow = p.change > 0 ? '↑' : p.change < 0 ? '↓' : '→';

  }).join('\n');
  ctx.replyWithHTML(`📡 Live Prices · SLOI AI\n\n${msg}`,
    Markup.inlineKeyboard([[Markup.button.webApp('📊 Full Exchange', MINI_APP_URL)]])
  );

bot.command('scan', bossOnly, async (ctx) => {

  await triggerLeadFinder();
  await triggerPriceScout();
  ctx.reply('✓ Agents running. Results will appear as messages.');
});

bot.command('buy', async (ctx) => {

// ── INLINE QUERY (any group) ─────────────────────────────────────────
bot.on('inline_query', async (ctx) => {
  const q = ctx.inlineQuery.query.toLowerCase();
  const products = await searchProducts(q);

  const results = products.map(p => ({
    type: 'article',
    id: p.ref,
    title: p.emoji + ' ' + p.name,
    description: `$${p.price.toLocaleString()}/${p.unit} · ${p.sector}`,
    input_message_content: {
      message_text:
        `${p.emoji} *${p.name}*\n` +
        `💰 *$${p.price.toLocaleString()}/${p.unit}* FOB\n` +
        `📦 MOQ: ${p.moq} ${p.unit} · Lead: ${p.lead_time_days} days\n\n` +
        `_via @SloiAIBot_`,
      parse_mode: 'Markdown',
    },
    reply_markup: Markup.inlineKeyboard([
      [Markup.button.webApp('🤖 Negotiate this price', MINI_APP_URL + `?product=${p.ref}`)],
      [Markup.button.webApp('📊 See order book', MINI_APP_URL)],
    ]).reply_markup,
  }));

  await ctx.answerInlineQuery(results, { cache_time: 30 });
});

// ── CALLBACK BUTTONS ─────────────────────────────────────────────────

// ── NATURAL LANGUAGE (Boss only) ──────────────────────────────────────
bot.on('text', bossOnly, async (ctx) => {
  const text = ctx.message.text;
  // Quick patterns
  if (/^approve$/i.test(text)) return bot.handleUpdate({ ...ctx.update, message: { ...ctx.message, text: '/approve' } });
  if (/^reject$/i.test(text) || /^דחה$/.test(text)) return bot.handleUpdate({ ...ctx.update, message: { ...ctx.message, text: '/reject' } });

  // Claude interprets natural language → API action
  const action = await claudeInterpret(text);
  if (action) {
    const result = await executeAction(action);
    ctx.reply(result.message);
  }
});

// Also notify Boss when AG-UI events fire
async function notifyBoss(event) {
  const alertTypes = ['AWAIT_HUMAN', 'LOI_GENERATED', 'NEGOTIATION_FAILED', 'COMPLIANCE_BLOCKED'];
  if (!alertTypes.includes(event.type)) return;

  const msg = formatTelegramAlert(event);
  await bot.telegram.sendMessage(process.env.TELEGRAM_BOSS_CHAT_ID, msg.text, msg.opts);
}

// Format AWAIT_HUMAN alert with inline buttons
function formatTelegramAlert(event) {
  if (event.type === 'AWAIT_HUMAN') {
    const d = event.data;
    return {
      text: `⚡ *APPROVAL NEEDED · ${event.session_id}*\n\n` +
            `💰 Agreed: *$${d.deal_price?.toLocaleString()}/MT*\n` +
            `📊 Total: *$${d.total_value?.toLocaleString()}*\n` +
            `📄 LOI fee: $${d.loi_fee || 500}\n` +
            `🔗 ${event.session_id}`,
      opts: {
        parse_mode: 'Markdown',
        reply_markup: {
          inline_keyboard: [[
            { text: '✅ Approve → LOI', callback_data: `approve_${event.session_id}` },
            { text: '✗ Reject', callback_data: `reject_${event.session_id}` },
          ]]
        }
      }
    };
  }
  // ... other event types
}

module.exports = { notifyBoss };

// Natural language interpreter (Claude)
NODE.JS — claudeInterpret()
async function claudeInterpret(text) {
  const res = await anthropic.messages.create({
    model: 'claude-sonnet-4-20250514',
    max_tokens: 200,
    system: `You are SLOI AI bot interpreter. Parse the Boss's message and return JSON action.
Available actions: approve, reject, status, prices, leads, scan, negotiate, balance
For negotiate: extract product, qty, target_price, max_price.
Return only JSON: {"action":"...", "params":{...}}
If unclear, return {"action":"unknown"}`,
    messages: [{ role: 'user', content: text }],
  });

  try {
    return JSON.parse(res.content[0].text);
  } catch {
    return null;
  }
}

async function executeAction(action) {
  switch(action.action) {
    case 'approve': {
      const negId = action.params?.neg_id || await getPendingNegId();
      await approveDeal(negId);
      return { message: `✅ Deal ${negId} approved. LOI generating...` };
    }
    case 'status': {
      const s = await getPlatformStatus();
      return { message: `📊 ${s.activeNegs} active · ${s.pendingApproval} pending · $${s.revenueMTD.toLocaleString()} MTD` };
    }
    case 'negotiate': {
      const neg = await startNegotiation(action.params);
      return { message: `🤖 Negotiation started · ${neg.id}\nI'll notify you when a deal is ready.` };
    }
    default:
      return { message: 'Use /help to see available commands.' };
  }
}
